# 中药正品与伪品差异化合物鉴别方案

## 应用场景

**问题**: 有 A 组（正品）和 B 组（伪品）两组 LC-MS/MS 数据，需要找出导致两组差异的关键化合物

**目标**:
1. 识别 A 组特有的化合物（正品标志物）
2. 识别 B 组特有的化合物（伪品标志物）
3. 识别两组共有但含量差异显著的化合物
4. 解析差异化合物的化学结构

---

## 完整分析流程

```
Step 1: 数据预处理与质控
   ↓
Step 2: DreaMS 嵌入向量生成
   ↓
Step 3: 分子网络构建 (A vs B)
   ↓
Step 4: 统计分析与差异筛选
   ↓
Step 5: 差异化合物鉴定 (三层鉴定)
   ↓
Step 6: 差异化合物验证与可视化
   ↓
Step 7: 化学分类与生物学解释
```

```python
#!/usr/bin/env python3
"""
script: 01_generate_dreams_embeddings.py
功能: 为 A、B 两组样本生成 DreaMS 嵌入向量
"""

import pandas as pd
from pathlib import Path
from core.embeddings_generator import generate_dreams_embeddings
from core.msp_parser import MSPParser

def merge_group_spectra(group_dir, output_msp):
    """合并一组的所有 MSP 文件"""
    all_spectra = []

    for msp_file in Path(group_dir).glob('*.msp'):
        parser = MSPParser(str(msp_file))
        spectra = parser.parse()

        # 添加样本信息
        sample_id = msp_file.stem
        for spectrum in spectra:
            spectrum['sample_id'] = sample_id

        all_spectra.extend(spectra)

    # 保存合并的 MSP
    save_merged_msp(all_spectra, output_msp)
    return all_spectra

def main():
    # 1. 合并 A 组谱图
    print("合并 A 组（正品）谱图...")
    spectra_A = merge_group_spectra(
        'input_data/group_A_authentic/',
        'results/merged_spectra_A.msp'
    )
    print(f"  A 组谱图数: {len(spectra_A)}")

    # 2. 合并 B 组谱图
    print("合并 B 组（伪品）谱图...")
    spectra_B = merge_group_spectra(
        'input_data/group_B_counterfeit/',
        'results/merged_spectra_B.msp'
    )
    print(f"  B 组谱图数: {len(spectra_B)}")

    # 3. 合并所有谱图用于统一分析
    print("合并 A+B 组所有谱图...")
    all_spectra = spectra_A + spectra_B
    save_merged_msp(all_spectra, 'results/merged_all.msp')
    print(f"  总谱图数: {len(all_spectra)}")

    # 4. 生成 DreaMS 嵌入向量
    print("\n生成 DreaMS 嵌入向量...")
    embeddings, _, _ = generate_dreams_embeddings(
        input_file='results/merged_all.msp',
        output_dir='results/embeddings/',
        output_format='npz',
        keep_mgf=True
    )
    print(f"  嵌入向量形状: {embeddings.shape}")

    # 5. 保存带标签的嵌入向量
    spectrum_info = []
    for i, spectrum in enumerate(all_spectra):
        spectrum_info.append({
            'index': i,
            'name': spectrum.get('name', f'spectrum_{i}'),
            'sample_id': spectrum.get('sample_id', 'unknown'),
            'group': 'A' if spectrum.get('sample_id', '').startswith('sample_A') else 'B',
            'precursor_mz': spectrum.get('precursor_mz', 0),
            'retention_time': spectrum.get('retention_time', 0)
        })

    df_info = pd.DataFrame(spectrum_info)
    df_info.to_csv('results/embeddings/spectrum_metadata.csv', index=False)

    print("\n✓ DreaMS 嵌入向量生成完成")

if __name__ == '__main__':
    main()
```

### 3.1 统计分析

```python
#!/usr/bin/env python3
"""
script: 02_differential_analysis.py
功能: 统计分析找出 A 和 B 组的差异化合物
"""

import numpy as np
import pandas as pd
from scipy import stats
from sklearn.decomposition import PCA
import matplotlib.pyplot as plt
import seaborn as sns

class DifferentialAnalyzer:
    """A vs B 差异分析"""

    def __init__(self, embeddings_file, metadata_file):
        # 加载数据
        self.embeddings = np.load(embeddings_file)['embeddings']
        self.metadata = pd.read_csv(metadata_file)

        # 分组索引
        self.group_A_idx = self.metadata[self.metadata['group'] == 'A'].index.tolist()
        self.group_B_idx = self.metadata[self.metadata['group'] == 'B'].index.tolist()

    def pca_analysis(self, output_dir='results/statistics/'):
        """PCA 降维分析，观察整体差异"""
        print("\n=== PCA 分析 ===")

        # PCA 降维
        pca = PCA(n_components=2)
        embeddings_pca = pca.fit_transform(self.embeddings)

        # 可视化
        plt.figure(figsize=(10, 8))

        # A 组
        plt.scatter(
            embeddings_pca[self.group_A_idx, 0],
            embeddings_pca[self.group_A_idx, 1],
            c='blue', label='Group A (Authentic)', alpha=0.6, s=50
        )

        # B 组
        plt.scatter(
            embeddings_pca[self.group_B_idx, 0],
            embeddings_pca[self.group_B_idx, 1],
            c='red', label='Group B (Counterfeit)', alpha=0.6, s=50
        )

        plt.xlabel(f'PC1 ({pca.explained_variance_ratio_[0]:.1%})')
        plt.ylabel(f'PC2 ({pca.explained_variance_ratio_[1]:.1%})')
        plt.title('PCA: Authentic vs Counterfeit')
        plt.legend()
        plt.grid(alpha=0.3)
        plt.savefig(f'{output_dir}/pca_plot.png', dpi=300, bbox_inches='tight')
        plt.close()

        print(f"  PC1 解释方差: {pca.explained_variance_ratio_[0]:.2%}")
        print(f"  PC2 解释方差: {pca.explained_variance_ratio_[1]:.2%}")
        print(f"  PCA 图保存: {output_dir}/pca_plot.png")

        return embeddings_pca

    def find_unique_compounds(self, similarity_threshold=0.85):
        """
        找出每组特有的化合物

        策略: 在自己组内相似度高，与对方组相似度低
        """
        print("\n=== 特有化合物筛选 ===")

        from scipy.spatial.distance import cosine

        unique_A = []  # A 组特有（正品标志物）
        unique_B = []  # B 组特有（伪品标志物）

        # 对每个谱图，计算与两组的平均相似度
        for i in range(len(self.embeddings)):
            emb = self.embeddings[i]

            # 与 A 组的平均相似度
            sim_to_A = np.mean([
                1 - cosine(emb, self.embeddings[j])
                for j in self.group_A_idx if j != i
            ])

            # 与 B 组的平均相似度
            sim_to_B = np.mean([
                1 - cosine(emb, self.embeddings[j])
                for j in self.group_B_idx if j != i
            ])

            # 判断是否为特有化合物
            if self.metadata.loc[i, 'group'] == 'A':
                if sim_to_A > similarity_threshold and sim_to_B < 0.7:
                    unique_A.append({
                        'index': i,
                        'name': self.metadata.loc[i, 'name'],
                        'sim_to_own_group': sim_to_A,
                        'sim_to_other_group': sim_to_B,
                        'specificity': sim_to_A - sim_to_B
                    })

            elif self.metadata.loc[i, 'group'] == 'B':
                if sim_to_B > similarity_threshold and sim_to_A < 0.7:
                    unique_B.append({
                        'index': i,
                        'name': self.metadata.loc[i, 'name'],
                        'sim_to_own_group': sim_to_B,
                        'sim_to_other_group': sim_to_A,
                        'specificity': sim_to_B - sim_to_A
                    })

        # 保存结果
        df_unique_A = pd.DataFrame(unique_A).sort_values('specificity', ascending=False)
        df_unique_B = pd.DataFrame(unique_B).sort_values('specificity', ascending=False)

        df_unique_A.to_csv('results/statistics/unique_to_authentic.csv', index=False)
        df_unique_B.to_csv('results/statistics/unique_to_counterfeit.csv', index=False)

        print(f"  A 组特有化合物（正品标志物）: {len(unique_A)} 个")
        print(f"  B 组特有化合物（伪品标志物）: {len(unique_B)} 个")

        return df_unique_A, df_unique_B

    def quantitative_difference(self, feature_table_A, feature_table_B,
                               fold_change_threshold=2.0, pvalue_threshold=0.05):
        """
        基于定量信息的差异分析

        需要 feature table 包含峰面积等定量信息
        """
        print("\n=== 定量差异分析 ===")

        # 读取特征量化表
        ft_A = pd.read_csv(feature_table_A)
        ft_B = pd.read_csv(feature_table_B)

        # 匹配化合物（基于 m/z 和 RT）
        matched_compounds = self.match_compounds_by_mz_rt(ft_A, ft_B)

        differential_compounds = []

        for compound in matched_compounds:
            # 获取两组的定量值
            intensities_A = compound['intensities_A']  # A 组各重复的强度
            intensities_B = compound['intensities_B']  # B 组各重复的强度

            # 统计检验 (t-test)
            t_stat, p_value = stats.ttest_ind(intensities_A, intensities_B)

            # 倍数变化
            mean_A = np.mean(intensities_A)
            mean_B = np.mean(intensities_B)
            fold_change = mean_A / mean_B if mean_B > 0 else np.inf

            # 筛选差异显著的化合物
            if abs(np.log2(fold_change)) > np.log2(fold_change_threshold) and p_value < pvalue_threshold:
                differential_compounds.append({
                    'compound': compound['name'],
                    'mz': compound['mz'],
                    'rt': compound['rt'],
                    'mean_intensity_A': mean_A,
                    'mean_intensity_B': mean_B,
                    'fold_change': fold_change,
                    'log2_fc': np.log2(fold_change),
                    'p_value': p_value,
                    'direction': 'upregulated_in_A' if fold_change > 1 else 'upregulated_in_B'
                })

        df_diff = pd.DataFrame(differential_compounds).sort_values('p_value')
        df_diff.to_csv('results/statistics/differential_compounds.csv', index=False)

        print(f"  差异显著化合物: {len(differential_compounds)} 个")
        print(f"    A 组上调: {sum(df_diff['direction'] == 'upregulated_in_A')} 个")
        print(f"    B 组上调: {sum(df_diff['direction'] == 'upregulated_in_B')} 个")

        # 火山图可视化
        self.plot_volcano(df_diff)

        return df_diff

    def plot_volcano(self, df_diff, output_dir='results/visualization/'):
        """绘制火山图"""
        plt.figure(figsize=(10, 8))

        # 所有点
        plt.scatter(
            df_diff['log2_fc'],
            -np.log10(df_diff['p_value']),
            c='gray', alpha=0.5, s=20
        )

        # 显著上调的点（A 组）
        up_A = df_diff[df_diff['direction'] == 'upregulated_in_A']
        plt.scatter(
            up_A['log2_fc'],
            -np.log10(up_A['p_value']),
            c='blue', alpha=0.8, s=50, label='Enriched in Authentic'
        )

        # 显著下调的点（B 组）
        up_B = df_diff[df_diff['direction'] == 'upregulated_in_B']
        plt.scatter(
            up_B['log2_fc'],
            -np.log10(up_B['p_value']),
            c='red', alpha=0.8, s=50, label='Enriched in Counterfeit'
        )

        # 阈值线
        plt.axhline(y=-np.log10(0.05), color='gray', linestyle='--', alpha=0.5)
        plt.axvline(x=1, color='gray', linestyle='--', alpha=0.5)
        plt.axvline(x=-1, color='gray', linestyle='--', alpha=0.5)

        plt.xlabel('log2(Fold Change)')
        plt.ylabel('-log10(p-value)')
        plt.title('Volcano Plot: Authentic vs Counterfeit')
        plt.legend()
        plt.grid(alpha=0.3)
        plt.savefig(f'{output_dir}/volcano_plot.png', dpi=300, bbox_inches='tight')
        plt.close()

        print(f"  火山图保存: {output_dir}/volcano_plot.png")

def main():
    analyzer = DifferentialAnalyzer(
        embeddings_file='results/embeddings/embeddings.npz',
        metadata_file='results/embeddings/spectrum_metadata.csv'
    )

    # 1. PCA 分析
    analyzer.pca_analysis()

    # 2. 特有化合物筛选
    unique_A, unique_B = analyzer.find_unique_compounds()

    # 3. 定量差异分析（如果有定量数据）
    if Path('input_data/group_A_authentic/feature_table_A.csv').exists():
        diff_compounds = analyzer.quantitative_difference(
            'input_data/group_A_authentic/feature_table_A.csv',
            'input_data/group_B_counterfeit/feature_table_B.csv'
        )

if __name__ == '__main__':
    main()
```

```python
#!/usr/bin/env python3
"""
script: 04_build_comparative_network.py
功能: 构建 A vs B 对比的分子网络
"""

from core.network_builder import NetworkBuilder
import pandas as pd
import networkx as nx

def build_comparative_network():
    """构建对比分子网络"""

    # 读取嵌入向量和元数据
    embeddings = np.load('results/embeddings/embeddings.npz')['embeddings']
    metadata = pd.read_csv('results/embeddings/spectrum_metadata.csv')

    # 构建网络
    builder = NetworkBuilder(
        embeddings=embeddings,
        spectrum_names=metadata['name'].tolist(),
        n_neighbors=5,
        similarity_threshold=0.85
    )

    G = builder.build_network()

    # 为节点添加分组信息
    for i, node in enumerate(G.nodes()):
        G.nodes[node]['group'] = metadata.loc[i, 'group']
        G.nodes[node]['sample_id'] = metadata.loc[i, 'sample_id']

    # 识别混合簇和特异性簇
    analyze_network_clusters(G, metadata)

    # 导出 GraphML（可用 Cytoscape 可视化）
    nx.write_graphml(G, 'results/networks/comparative_network.graphml')

    print("✓ 对比分子网络构建完成")
    print(f"  节点数: {G.number_of_nodes()}")
    print(f"  边数: {G.number_of_edges()}")
    print(f"  网络文件: results/networks/comparative_network.graphml")

def analyze_network_clusters(G, metadata):
    """分析网络簇的组成"""

    # 社区检测
    from networkx.algorithms import community
    communities = community.greedy_modularity_communities(G)

    cluster_analysis = []

    for i, comm in enumerate(communities):
        # 统计该簇中 A 和 B 组的比例
        nodes_in_comm = list(comm)
        groups = [G.nodes[node]['group'] for node in nodes_in_comm]

        count_A = groups.count('A')
        count_B = groups.count('B')
        total = len(nodes_in_comm)

        cluster_type = None
        if count_A / total > 0.8:
            cluster_type = 'A-specific (正品特异性簇)'
        elif count_B / total > 0.8:
            cluster_type = 'B-specific (伪品特异性簇)'
        else:
            cluster_type = 'Mixed (混合簇)'

        cluster_analysis.append({
            'cluster_id': i,
            'size': total,
            'count_A': count_A,
            'count_B': count_B,
            'ratio_A': count_A / total,
            'ratio_B': count_B / total,
            'cluster_type': cluster_type
        })

        # 为节点添加簇 ID
        for node in nodes_in_comm:
            G.nodes[node]['cluster_id'] = i
            G.nodes[node]['cluster_type'] = cluster_type

    # 保存簇分析结果
    df_clusters = pd.DataFrame(cluster_analysis)
    df_clusters.to_csv('results/networks/cluster_analysis.csv', index=False)

    print(f"\n簇分析:")
    print(f"  总簇数: {len(communities)}")
    print(f"  A 组特异性簇: {sum(df_clusters['cluster_type'] == 'A-specific (正品特异性簇)')}")
    print(f"  B 组特异性簇: {sum(df_clusters['cluster_type'] == 'B-specific (伪品特异性簇)')}")
    print(f"  混合簇: {sum(df_clusters['cluster_type'] == 'Mixed (混合簇)')}")

if __name__ == '__main__':
    build_comparative_network()
```

```python
import matplotlib.pyplot as plt

from collections import Counter

def analyze_chemical_classes(identified_df):
    """分析化学分类"""
    from pymongo import MongoClient

    # 连接 LOTUS MongoDB
    client = MongoClient()
    db = client['lotus']
    collection = db['lotusUniqueNaturalProduct']

    classes_authentic = []
    classes_counterfeit = []

    for _, row in identified_df.iterrows():
        # 通过 SMILES 查询 LOTUS
        smiles = row['smiles']
        lotus_entry = collection.find_one({'smiles': smiles})

        if lotus_entry:
            chem_class = lotus_entry.get('chemicalTaxonomyNPclassifierClass', 'Unknown')

            if row['differential_type'] == 'Authentic-specific (正品特有)':
                classes_authentic.append(chem_class)
            elif row['differential_type'] == 'Counterfeit-specific (伪品特有)':
                classes_counterfeit.append(chem_class)

    return {
        'authentic': Counter(classes_authentic),
        'counterfeit': Counter(classes_counterfeit)
    }
```
